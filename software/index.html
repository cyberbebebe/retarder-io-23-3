# RESTfull сервіс

from fastapi import FastAPI, Body, HTTPException, Depends
from pydantic import BaseModel
import mysql.connector
import passlib
import uuid


mydb = mysql.connector.connect(
    host="localhost",
    user="root",
    password="admin",
    database="mydb"
)

mycursor = mydb.cursor()

app = FastAPI()


class Client(BaseModel):
    id: str
    login: str


@app.get("/clients")
async def clients_list():
    try:
        sql = "SELECT id, login FROM client"
        mycursor.execute(sql)
        result = mycursor.fetchall()

        clients = [Client(id=row[0], login=row[1]) for row in result]

        return clients

    except mysql.connector.Error as e:
        print(f"Error during fetching clients: {e}")
        raise HTTPException(status_code=500, detail="Internal server error")

    except Exception as e:
        print(f"Unexpected error: {e}")
        raise HTTPException(status_code=500, detail="Internal server error")


class User(BaseModel):
    login: str
    email: str
    password: str
    role_id: int = 2


@app.post("/register")
async def register(user: User):
    try:
        sql = "SELECT * FROM client WHERE login = %s OR email = %s"
        val = (user.login, user.email)
        mycursor.execute(sql, val)
        result = mycursor.fetchall()

        if result:
            raise HTTPException(
                status_code=400, detail="Username or email already exists")

        user_id = str(uuid.uuid4())
        sql = "INSERT INTO client (id, login, email, password, role_id) VALUES (%s, %s, %s, %s, 2)"
        val = (user_id, user.login, user.email, user.password)
        mycursor.execute(sql, val)
        mydb.commit()

        return {"message": "Registered successfully!"}

    except Exception as e:
        print(f"Error registering user: {e}")
        raise HTTPException(
            status_code=400, detail=f"Registration failed, {e}")


class LoginCredentials(BaseModel):
    login: str
    password: str


@app.post("/login")
async def login(credentials: LoginCredentials):
    try:
        sql = "SELECT id, login, password FROM client WHERE login = %s"
        val = (credentials.login,)
        mycursor.execute(sql, val)
        result = mycursor.fetchone()

        if not result or result[2] != credentials.password:
            raise HTTPException(status_code=401, detail=f"Invalid credentials")

        login = result[1]
        return {"message": f"Welcome, {login}"}

    except Exception as e:
        print(f"Error during login: {e}")
        raise HTTPException(
            status_code=400, detail=f"Login failed")


class ChangePassword(BaseModel):
    login: str
    old_password: str
    new_password: str


@app.put("/change-password")
async def change_password(change_data: ChangePassword):
    try:
        sql = "SELECT id, password FROM client WHERE login = %s"
        val = (change_data.login,)
        mycursor.execute(sql, val)
        result = mycursor.fetchone()

        if not result:
            raise HTTPException(status_code=401, detail="Invalid credentials")

        if change_data.old_password != result[1]:
            raise HTTPException(
                status_code=401, detail="Incorrect old password")

        sql = "UPDATE client SET password = %s WHERE id = %s"
        val = (change_data.new_password, result[0])
        mycursor.execute(sql, val)
        mydb.commit()

        return {"message": "Password changed successfully"}

    except Exception as e:
        print(f"Error during password change: {e}")
        raise HTTPException(status_code=400, detail="Password change failed")


class MediaData(BaseModel):
    name: str
    filetype: str
    metadata: str


@app.post("/data")
async def add_media(media_data: MediaData):
    try:

        sql = "INSERT INTO mediadata (name, filetype, metadata) VALUES (%s, %s, %s)"
        val = (media_data.name, media_data.filetype, media_data.metadata)
        mycursor.execute(sql, val)
        mydb.commit()

        return {"message": "Media data added successfully"}

    except Exception as e:
        print(f"Error during adding media: {e}")
        raise HTTPException(status_code=400, detail="Adding media failed")


def application_shutdown():
    mydb.close()


@app.delete("/data")
async def delete_media(media_data: MediaData):
    try:
        sql = "DELETE FROM mediadata WHERE name = %s AND filetype = %s AND metadata = %s"
        val = (media_data.name, media_data.filetype, media_data.metadata)
        mycursor.execute(sql, val)
        mydb.commit()

        affected_rows = mycursor.rowcount

        if affected_rows == 0:
            raise HTTPException(status_code=404, detail="Media data not found")

        return {"message": "Media data deleted successfully"}

    except Exception as e:
        print(f"Error during deleting media: {e}")
        raise HTTPException(status_code=400, detail="Deleting media failed")


app.add_event_handler("shutdown", application_shutdown)
